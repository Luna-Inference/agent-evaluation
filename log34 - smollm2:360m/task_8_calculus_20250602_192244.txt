def integrate_sin_cos(a, b):                                                                                                                                                     
    # Calculate the limits of integration                                                                                                                                        
    x1 = np.linspace(a, b, 50)                                                                                                                                                   
    y1 = np.sin(x1 * np.pi / (2 * np.pi))                                                                                                                                        
    z1 = np.cos(x1 * np.pi / (2 * np.pi))                                                                                                                                        
                                                                                                                                                                                 
    # Calculate the definite integral                                                                                                                                            
    integral_result = np.trapz(y1, x1)                                                                                                                                           
                                                                                                                                                                                 
    return integral_result
