import math

def volume_of_solid(a, b):                                                                                                                                                       
    """                                                                                                                                                                          
    Calculate the volume of a solid formed by rotating y = ln(x) around the x-axis from x=1 to x=e.                                                                              
                                                                                                                                                                                 
    Args:                                                                                                                                                                        
        a (float): Lower limit of integration for y = ln(x).                                                                                                                     
        b (float): Upper limit of integration for y = ln(x).                                                                                                                     
                                                                                                                                                                                 
    Returns:                                                                                                                                                                     
        float: The volume of the solid.                                                                                                                                          
    """                                                                                                                                                                          
    return special.pochhammer(b, 2) * math.sqrt(special.gamma(a+1))                                                                                                              
                                                                                                                                                                                 
volume = volume_of_solid(1, e)                                                                                                                                                   
print(volume)
