from collections import Counter                                                                                                                                                  
                                                                                                                                                                                 
                                                                                                                                                                                 
def search(query_string, word):                                                                                                                                                  
    """Returns the word from the given query and its count in the counter."""                                                                                                    
                                                                                                                                                                                 
    # Use regular expression for pattern matching                                                                                                                                
    pattern = r"(\w+)"  # Search all words that begin with 'word'                                                                                                                
    match = re.search(pattern, query_string)                                                                                                                                     
                                                                                                                                                                                 
    if not match:                                                                                                                                                                
        raise ValueError("No matches found")                                                                                                                                     
                                                                                                                                                                                 
    return word[:match.start()] + " ".join([word[i:] for i in range(len(match.group()))])
