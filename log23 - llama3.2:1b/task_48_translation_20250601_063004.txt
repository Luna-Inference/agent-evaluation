    phrases = ['hola, ¿cómo?', '¿cómo estás?', 'dónde está...', 'cuánto cuesta?', 'dónde puedo encontrar...']
    sentence = text

    for phrase in phrases:
        if phrase.lower() not in spanish_translation:
            print(f"'{phrase}' is not found in the Spanish translation dictionary.")
            return f"{' '.join([k for k, v in spanish_translation.items() if v == phrase.lower()])}"

    # Translate the sentence using a language model
    translated_text = ''

        if phrases[i].lower() not in spanish_translation:

            return f"{' '.join([k for k, v in spanish_translation.items() if v == phrases[i]])}"
        else:

    print(f"The following sentence was translated from '{text}': {translated_text.strip()}")
