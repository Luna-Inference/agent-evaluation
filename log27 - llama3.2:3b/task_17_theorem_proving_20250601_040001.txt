import math                                                                                                                                                  
                                                                                                                                                             
# Function to check if the ratio of two integers is irrational                                                                                               
def check_ratio():                                                                                                                                           
    sqrt_2 = math.sqrt(2)                                                                                                                                    
                                                                                                                                                             
    # Check if √2 can be expressed as a ratio of two integers p and q                                                                                        
    for i in range(1, 10000):                                                                                                                                
        numerator = i                                                                                                                                        
        denominator = 1                                                                                                                                      
                                                                                                                                                             
        if numerator != int(numerator) :                                                                                                                     
            rational_sqrt_2 = (numerator / denominator) ** 2                                                                                                 
                                                                                                                                                             
            # Check if the ratio is in its simplest form                                                                                                     
            while True:                                                                                                                                      
                gcd = math.gcd(int(rational_sqrt_2), denominator)                                                                                            
                if gcd == 1:                                                                                                                                 
                    break                                                                                                                                    
                                                                                                                                                             
                rational_sqrt_2 /= gcd                                                                                                                       
                denominator /= gcd                                                                                                                           
                                                                                                                                                             
            if abs(rational_sqrt_2 - sqrt_2) < 0.000001:                                                                                                     
                return False                                                                                                                                 
                                                                                                                                                             
    # If no contradiction is found, assume that √2 is irrational                                                                                             
    print("No contradiction found. Assuming √2 is irrational")

    return True                                                                                                                                              
                                                                                                                                                             
check_ratio()
