panel_length_ft = span_ft / num_panels                                                                                                                                           
                                                                                                                                                                                 
# List node positions

members = []                                                                                                                                                                     
                                                                                                                                                                                 
# Bottom chord                                                                                                                                                                   
for i in range(num_panels):                                                                                                                                                      
    members.append(('bottom_chord', nodes[i], nodes[i+1]))                                                                                                                       
# Top chord                                                                                                                                                                      
for i in range(num_panels-1):                                                                                                                                                    
    members.append(('top_chord', top_nodes[i], top_nodes[i+1]))                                                                                                                  
# Verticals                                                                                                                                                                      
for i in range(1, num_panels):                                                                                                                                                   
    members.append(('vertical', nodes[i], top_nodes[i-1]))                                                                                                                       
# Diagonals (Pratt: only one diagonal per panel, direction alternates)                                                                                                           
for i in range(num_panels):                                                                                                                                                      
    if i < len(top_nodes):                                                                                                                                                       
        members.append(('diagonal', nodes[i], top_nodes[i]))                                                                                                                     
    if i > 0:                                                                                                                                                                    
        members.append(('diagonal', nodes[i+1], top_nodes[i-1]))                                                                                                                 
                                                                                                                                                                                 
# Output truss design                                                                                                                                                            
print(f"Truss Type: Pratt Truss")                                                                                                                                                
print(f"Span: {span_ft} ft  |  Panels: {num_panels}  |  Panel Length: {panel_length_ft} ft")                                                                                     
print("\nNodes (coordinates in ft):")                                                                                                                                            
for idx, pt in enumerate(nodes):                                                                                                                                                 
    print(f"  Bottom Node {idx+1}: {pt}")                                                                                                                                        
for idx, pt in enumerate(top_nodes):                                                                                                                                             
    print(f"  Top Node {idx+1}: {pt}")                                                                                                                                           
                                                                                                                                                                                 
print("\nMembers (type, from, to):")

print(f"\nEstimated maximum member force (for sizing): {max_member_force:.0f} lbs")                                                                                              
                                                                                                                                                                                 
# Recommendation for member sizing (conservative): use steel angle or tube, minimum yield > {max_member_force:.0f} lbs

print("This is a simplified estimate; actual member size must be checked according to AISC or NDS standards for deflection, buckling, and safety.")
