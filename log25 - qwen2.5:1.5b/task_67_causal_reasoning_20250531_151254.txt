import matplotlib.pyplot as plt

plt.show()                                                                                                                                                                       
                                                                                                                                                                                 
# Perform linear regression analysis to find the causal relationship between variables                                                                                           
X = data[['GDP']]                                                                                                                                                                
y = data[['Unemployment Rate']]                                                                                                                                                  
                                                                                                                                                                                 
model = LinearRegression()                                                                                                                                                       
model.fit(X, y)                                                                                                                                                                  
                                                                                                                                                                                 
print(f"The slope of the line is: {model.coef_[0]}")                                                                                                                             
print(f"The intercept of the line is: {model.intercept_}")                                                                                                                       
                                                                                                                                                                                 
# Print a summary of the regression model                                                                                                                                        
print(model.summary())                                                                                                                                                           
                                                                                                                                                                                 
# Check for other variables that might have affected the financial crisis                                                                                                        
plt.figure(figsize=(10, 5))                                                                                                                                                      
for variable in data.columns:                                                                                                                                                    
    if variable != 'GDP' and variable != 'Unemployment Rate':                                                                                                                    
        plt.scatter(data['Year'], data[variable])                                                                                                                                
        plt.xlabel('Year')                                                                                                                                                       
        plt.ylabel(variable)                                                                                                                                                     
        plt.title(f'Scatter plot of {variable} over Time')                                                                                                                       
        plt.show()                                                                                                                                                               
                                                                                                                                                                                 
# The interconnections between the variables can be inferred from these scatter plots.
