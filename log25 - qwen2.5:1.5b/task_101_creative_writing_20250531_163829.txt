while True:                                                                                                                                                                      
    action = input("What would you like to do?         while True:                                                                                                                                                              
            question = input("Question (AI/Other): ").strip().lower()

                answer = "Advocate: Of course, AI is capable of analyzing data faster than humans can and it can process information in ways that are too complex for us to      
understand."                                                                                                                                                                     
                dialogue.append(f"{answer}")                                                                                                                                     
                                                                                                                                                                                 
            elif question == "Other":                                                                                                                                            
                answer = input("Please ask a valid question about AI (e.g., 'What's the potential benefits?').: ").strip().lower()                                               
                if answer:                                                                                                                                                       
                    dialogue.append(f"Critic: While Advocate is right, we need to be cautious of how AI could potentially harm us. For example, there have been instances where  
self-driving cars caused accidents."                                                                                                                                             
                           f" We also need to consider the ethical implications.")                                                                                               
                    break                                                                                                                                                        
                                                                                                                                                                                 
            else:                                                                                                                                                                
                print("Invalid question. Please try again.")                                                                                                                     
                                                                                                                                                                                 
    elif action == "Critic":                                                                                                                                                     
        dialogue.append(input("Critic: You're correct about some of the benefits of AI, but I believe it's important to be cautious."                                            
                               f" For instance, there are concerns over job displacement and how AI could replace human workers.")                                               
                                                                                                                                                                                 
        while True:                                                                                                                                                              
            question = input("Question (AI/Other): ").strip().lower()                                                                                                            
                                                                                                                                                                                 
            if question.startswith("ai"):                                                                                                                                        
                answer = "Critic: You're right about the potential benefits of AI. However, it's important to consider the ethical implications as well."

                answer = input("Please ask a valid question about AI (e.g., 'What are the pros?').:")                                                                            
                if answer:                                                                                                                                                       
                    dialogue.append(f"Critic: While you're right about some of its benefits, we need to be aware of the potential negative consequences. For example, there have 
been instances where self-driving cars caused accidents."                                                                                                                        
                               f" We also need to consider the ethical implications.")

                print("Invalid question. Please try again.")                                                                                                                     
                                                                                                                                                                                 
    elif action == "quit":                                                                                                                                                       
        break
